pub fn contains_tile(&self, tile: TileId) -> bool

Check whether the edition contains a specific tile.

Returns true when the identifier is found in the list of tiles.

Compares the full tile identifier, including the graphical variant. To compare the base identifier only use Edition::contains_base_tile() instead. See TileId::base() for more information on the difference between base and full tile identifiers.

§Examples
let edition = Edition::DirtCoreBox;
assert_eq!(edition.contains_tile(tile!(201, a)), true);
assert_eq!(edition.contains_tile(tile!(325, b)), false);

let edition = Edition::DirtClimb;
assert_eq!(edition.contains_tile(tile!(201, a)), false);
assert_eq!(edition.contains_tile(tile!(325, b)), true);

let edition = Edition::GtWorldTour;
assert_eq!(edition.contains_tile(tile!(124, a, 1)), false);
assert_eq!(edition.contains_tile(tile!(124, a, 2)), true);

pub fn contains_base_tile(&self, tile: TileId) -> bool

Check whether the edition matches a given base tile identifier.

Returns true when a tile with matching base identifier is found in the list of tiles.

Compares the base tile identifier, with graphical variant excluded. To compare the full identifier use Edition::contains_tile() instead. See TileId::base() for more information on the difference between base and full tile identifiers.

§Examples
let edition = Edition::DirtCoreBox;
assert_eq!(edition.contains_base_tile(tile!(201, a)), true);
assert_eq!(edition.contains_base_tile(tile!(325, b)), false);

let edition = Edition::DirtClimb;
assert_eq!(edition.contains_base_tile(tile!(201, a)), false);
assert_eq!(edition.contains_base_tile(tile!(325, b)), true);

let edition = Edition::GtWorldTour;
assert_eq!(edition.contains_base_tile(tile!(124, a, 1).base()), true);
assert_eq!(edition.contains_base_tile(tile!(124, a, 2).base()), true);
