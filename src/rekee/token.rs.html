22/// Identifier for a token which can be placed on tiles.
23///
24/// # Examples
25///
26/// ```
27/// # use rekee::token::*;
28/// # use rekee::tile::Terrain;
29/// let token: TokenId = "chicane".parse().unwrap();
30/// assert_eq!(token, TokenId::Chicane(Terrain::None));
31///
32/// let token = TokenId::Jump(Terrain::Gravel);
33/// assert_eq!(token.to_string(), "Jump Gravel");
34/// ```
35#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord)]
36#[derive(Deserialize)]
37#[serde(try_from = "&str")]
38pub enum TokenId {
39    /// Chicane token without a speed limit, part of the Rallyman DIRT Copilot Pack.
40    Chicane(Terrain),
41    /// Chicane token with a speed limit, part of the Rallyman DIRT Copilot Pack.
42    ChicaneWithLimit(Terrain),
43    /// Jump token, part of the Rallyman DIRT Copilot Pack.
44    Jump(Terrain),
45    /// Water token, part of the Rallyman DIRT Copilot Pack.
46    Water(Terrain),
47    /// Ascent tile modifier, part of the Climb expansion for Rallyman DIRT.
48    ClimbAscent,
49    /// Descent tile modifier, part of the Climb expansion for Rallyman DIRT.
50    ClimbDescent,
51    /// Cloud tile modifier, part of the Climb expansion for Rallyman DIRT.
52    Cloud,
53    /// Oxygen tile modifier, part of the Climb expansion for Rallyman DIRT.
54    Oxygen(u8),
55    /// Joker entrance arch, part of the RX expansion for Rallyman DIRT.
56    JokerEntrance,
57    /// Joker exit arch, part of the RX expansion for Rallyman DIRT.
58    JokerExit,
59    /// Finishing line arch, part of the RX expansion for Rallyman DIRT.
60    Finish,
61    /// Shortcut mud spray, part of the Rallyman DIRT core box.
62    MudSpray,
63}
64
65impl TokenId {
66    /// Creates a copy of the current token with inner terrain changed to a
67    /// given value.
68    ///
69    /// # Examples
70    ///
71    /// ```
72    /// # use rekee::token::*;
73    /// # use rekee::tile::Terrain;
74    /// let token = TokenId::Chicane(Terrain::Asphalt);
75    /// let other_token = token.with_terrain(Terrain::Snow);
76    /// assert_eq!(other_token, TokenId::Chicane(Terrain::Snow));
77    /// ```
78    pub fn with_terrain(&self, terrain: Terrain) -> Self {
79        match self {
80            TokenId::Chicane(_) =>
81                TokenId::Chicane(terrain),
82            TokenId::ChicaneWithLimit(_) =>
83                TokenId::ChicaneWithLimit(terrain),
84            TokenId::Jump(_) =>
85                TokenId::Jump(terrain),
86            TokenId::Water(_) =>
87                TokenId::Water(terrain),
88            _ =>
89                *self
90        }
91    }
92
93    #[deprecated(note = "Please use TokenId::base() instead")]
94    pub fn without_terrain(&self) -> Self {
95        self.with_terrain(Terrain::None)
96    }
97
98    /// Creates a copy of the current token with inner oxygen number changed to
99    /// the given value.
100    pub(crate) fn with_number(&self, number: u8) -> Self {
101        match self {
102            TokenId::Oxygen(_) =>
103                TokenId::Oxygen(number),
104            _ =>
105                *self
106        }
107    }
108
109    /// Creates a copy of the current token without terrain info or oxygen level.
110    pub fn base(&self) -> Self {
111        match self {
112            TokenId::Chicane(_) =>
113                TokenId::Chicane(Terrain::None),
114            TokenId::ChicaneWithLimit(_) =>
115                TokenId::ChicaneWithLimit(Terrain::None),
116            TokenId::Jump(_) =>
117                TokenId::Jump(Terrain::None),
118            TokenId::Water(_) =>
119                TokenId::Water(Terrain::None),
120            TokenId::Oxygen(_) =>
121                TokenId::Oxygen(0),
122            _ =>
123                *self
124        }
125    }
126
127    /// Iterator over all track tokens.
128    ///
129    /// Tokens with terrain use the neutral `Terrain::None` value.
130    ///
131    /// # Examples
132    ///
133    /// ```
134    /// # use rekee::token::*;
135    /// let tokens: Vec<String> = TokenId::iter()
136    ///     .map(|val| val.to_string())
137    ///     .collect();
138    /// assert_eq!(tokens, vec![
139    ///     "Chicane",
140    ///     "Chicane Limit",
141    ///     "Jump",
142    ///     "Water",
143    ///     "Climb Ascent",
144    ///     "Climb Descent",
145    ///     "Cloud",
146    ///     "Oxygen",
147    ///     "Joker Entrance",
148    ///     "Joker Exit",
149    ///     "Finish",
150    ///     "Mud Spray"]);
151    /// ```
152    pub fn iter() -> std::slice::Iter<'static, Self> {
153        const TOKENS: [TokenId; 12] = [
154            TokenId::Chicane(Terrain::None),
155            TokenId::ChicaneWithLimit(Terrain::None),
156            TokenId::Jump(Terrain::None),
157            TokenId::Water(Terrain::None),
158            TokenId::ClimbAscent,
159            TokenId::ClimbDescent,
160            TokenId::Cloud,
161            TokenId::Oxygen(0),
162            TokenId::JokerEntrance,
163            TokenId::JokerExit,
164            TokenId::Finish,
165            TokenId::MudSpray,
166        ];
167        TOKENS.iter()
168    }
169
170    #[deprecated(note = "Please use Token::editions() instead")]
171    pub fn edition(&self) -> Edition {
172        *self.editions().first().unwrap()
173    }
174
175    /// List of editions which contain this token.
176    ///
177    /// The returned list will contain a single edition entry for most tokens,
178    /// but an additional entry for some rare cases.
179    ///
180    /// # Examples
181    ///
182    /// ```
183    /// # use rekee::edition::Edition;
184    /// # use rekee::tile::Terrain;
185    /// # use rekee::token::TokenId;
186    /// let token = TokenId::Jump(Terrain::Asphalt);
187    /// assert_eq!(token.editions(), vec![Edition::DirtMonteCarlo, Edition::DirtCopilotPack]);
188    ///
189    /// let token = TokenId::JokerEntrance;
190    /// assert_eq!(token.editions(), vec![Edition::DirtRx]);
191    /// ```
192    pub fn editions(&self) -> Vec<Edition> {
193        match self {
194            TokenId::Chicane(_) =>
195                [Edition::DirtMonteCarlo, Edition::DirtCopilotPack].to_vec(),
196            TokenId::ChicaneWithLimit(_) =>
197                [Edition::DirtMonteCarlo, Edition::DirtCopilotPack].to_vec(),
198            TokenId::Jump(_) =>
199                [Edition::DirtMonteCarlo, Edition::DirtCopilotPack].to_vec(),
200            TokenId::Water(_) =>
201                [Edition::DirtMonteCarlo, Edition::DirtCopilotPack].to_vec(),
202            TokenId::ClimbAscent =>
203                [Edition::DirtClimb].to_vec(),
204            TokenId::ClimbDescent =>
205                [Edition::DirtClimb].to_vec(),
206            TokenId::Cloud =>
207                [Edition::DirtClimb].to_vec(),
208            TokenId::Oxygen(_) =>
209                [Edition::DirtClimb].to_vec(),
210            TokenId::JokerEntrance =>
211                [Edition::DirtRx].to_vec(),
212            TokenId::JokerExit =>
213                [Edition::DirtRx].to_vec(),
214            TokenId::Finish =>
215                [Edition::DirtRx].to_vec(),
216            TokenId::MudSpray =>
217                [Edition::DirtCoreBox].to_vec(),
218        }
219    }
220}

228impl fmt::Display for TokenId {
229    fn fmt(&self, fmt: &mut fmt::Formatter) -> fmt::Result {
230        let mut terrain = Terrain::None;
231        match self {
232            TokenId::Chicane(val) => {
233                terrain = *val;
234                write!(fmt, "Chicane")?;
235            },
236            TokenId::ChicaneWithLimit(val) => {
237                terrain = *val;
238                write!(fmt, "Chicane Limit")?;
239            },
240            TokenId::Jump(val) => {
241                terrain = *val;
242                write!(fmt, "Jump")?;
243            },
244            TokenId::Water(val) => {
245                terrain = *val;
246                write!(fmt, "Water")?;
247            },
248            TokenId::ClimbAscent =>
249                write!(fmt, "Climb Ascent")?,
250            TokenId::ClimbDescent =>
251                write!(fmt, "Climb Descent")?,
252            TokenId::Cloud =>
253                write!(fmt, "Cloud")?,
254            TokenId::Oxygen(val) => {
255                write!(fmt, "Oxygen")?;
256                if *val > 0 {
257                    write!(fmt, " {}", val)?;
258                }
259            }
260            TokenId::JokerEntrance =>
261                write!(fmt, "Joker Entrance")?,
262            TokenId::JokerExit =>
263                write!(fmt, "Joker Exit")?,
264            TokenId::Finish =>
265                write!(fmt, "Finish")?,
266            TokenId::MudSpray =>
267                write!(fmt, "Mud Spray")?,
268        };
269        if terrain != Terrain::None {
270            write!(fmt, " {}", terrain)?;
271        }
272        Ok(())
273    }

285impl FromStr for TokenId {
286    type Err = ParseTokenError;
287
288    fn from_str(val: &str) -> Result<Self, Self::Err> {
289        // match strings from both trait implementations, std::fmt::Display and serde::Serialize
290        let mut s = val.replace(char::is_whitespace, "-");
291        s.make_ascii_lowercase();
292
293        let mut name = s.as_ref();
294        let mut terrain = Terrain::None;
295        let mut number = 0_u8;
296        if let Some((prefix, suffix)) = s.rsplit_once('-') {
297            if let Ok(val) = suffix.parse() {
298                terrain = val;
299                name = prefix;
300            }
301            if let Ok(val) = suffix.parse() {
302                number = val;
303                name = prefix;
304            }
305        }
306
307        match name {
308            "chicane" =>
309                Ok(TokenId::Chicane(terrain)),
310            "chicane-limit" =>
311                Ok(TokenId::ChicaneWithLimit(terrain)),
312            "jump" =>
313                Ok(TokenId::Jump(terrain)),
314            "water" =>
315                Ok(TokenId::Water(terrain)),
316            "climb-ascent" =>
317                Ok(TokenId::ClimbAscent),
318            "climb-descent" =>
319                Ok(TokenId::ClimbDescent),
320            "cloud" =>
321                Ok(TokenId::Cloud),
322            "oxygen" =>
323                Ok(TokenId::Oxygen(number)),
324            "joker-entrance" =>
325                Ok(TokenId::JokerEntrance),
326            "joker-exit" =>
327                Ok(TokenId::JokerExit),
328            "finish" =>
329                Ok(TokenId::Finish),
330            "mud-spray" =>
331                Ok(TokenId::MudSpray),
332            _ =>
333                Err(ParseTokenError::UnknownName(name.to_string())),
334        }
335    }
336}

341}

343impl fmt::Display for ParseTokenError {
344    fn fmt(&self, fmt: &mut fmt::Formatter) -> fmt::Result {
345        match self {
346            ParseTokenError::UnknownName(val) =>
347                write!(fmt, "Unknown token name \"{}\"", val),
348        }
349    }

420/// Utility methods on any list of [`TokenId`]s.
421pub trait TokenList {
422
423    /// Returns a summary about the tokens that are contained within the current
424    /// list of tokens.
425    ///
426    /// For each token the entry contains the number of instances that occur in
427    /// the current list of tiles. Note that terrain information is ignored when
428    /// building the token summary. The returned array is always sorted by token
429    /// identifier.
430    ///
431    /// # Examples
432    ///
433    /// ```
434    /// # use rekee::tile::Terrain;
435    /// # use rekee::token::{TokenId, TokenList};
436    /// let tokens = vec![
437    ///     TokenId::JokerEntrance,
438    ///     TokenId::Chicane(Terrain::Gravel),
439    ///     TokenId::Chicane(Terrain::Asphalt),
440    ///     TokenId::Oxygen(1),
441    /// ];
442    /// for row in tokens.token_summary() {
443    ///     println!("{}x token {}", row.count, row.token);
444    /// }
445    /// ```
446    fn token_summary(&self) -> Vec<TokenSummary>;
447
448    /// Returns a summary about the editions that are necessary to build the
449    /// current list of tokens.
450    ///
451    /// For each edition the entry contains the number of tokens that belong to
452    /// the edition.
453    ///
454    /// # Examples
455    ///
456    /// ```
457    /// # use rekee::tile::Terrain;
458    /// # use rekee::token::{TokenId, TokenList};
459    /// let tokens = vec![
460    ///     TokenId::JokerEntrance,
461    ///     TokenId::Chicane(Terrain::Gravel),
462    ///     TokenId::Chicane(Terrain::Asphalt),
463    ///     TokenId::Oxygen(1),
464    /// ];
465    /// for row in tokens.edition_summary() {
466    ///     println!("{}: {} tokens", row.edition, row.token_count);
467    /// }
468    /// ```
469    fn edition_summary(&self) -> Vec<EditionSummary>;
470}
471
472impl<T: AsRef<TokenId> + Clone> TokenList for [T] {
473    fn token_summary(&self) -> Vec<TokenSummary> {
474        let mut summary = BTreeMap::new();
475        for token in self.iter() {
476            let base_token = token.as_ref()
477                .with_terrain(Terrain::None);
478            let entry = summary.entry(base_token)
479                .or_insert(0);
480            *entry += 1;
481        }
482        let summary: Vec<_> = summary.iter()
483            .map(|(&token, &count)|
484                TokenSummary::new(token, count)
485            )
486            .collect();
487        summary
488    }
489
490    fn edition_summary(&self) -> Vec<EditionSummary> {
491        let mut summary = BTreeMap::new();
492        for token in self.iter() {
493            // We restrict edition lookup to the first/main entry here
494            let edition = match token.as_ref().editions().first() {
495                Some(val) => *val,
496                None => continue,
497            };
498            let entry = summary.entry(edition)
499                .or_insert(0);
500            *entry += 1;
501        }
502        let summary: Vec<_> = summary.iter()
503            .map(|(&edition, &count)|
504                EditionSummary::new(edition, count)
505            )
506            .collect();
507        summary
508    }
509}
