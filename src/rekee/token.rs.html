22/// Identifier for a token which can be placed on tiles.
23///
24/// # Examples
25///
26/// ```
27/// # use rekee::token::*;
28/// # use rekee::tile::Terrain;
29/// let token: TokenId = "chicane".parse().unwrap();
30/// assert_eq!(token, TokenId::Chicane(Terrain::None));
31///
32/// let token = TokenId::Jump(Terrain::Gravel);
33/// assert_eq!(token.to_string(), "Jump Gravel");
34/// ```
35#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord)]
36#[derive(Deserialize)]
37#[serde(try_from = "&str")]
38pub enum TokenId {
39    /// Chicane token without a speed limit, part of the Rallyman DIRT Copilot Pack.
40    Chicane(Terrain),
41    /// Chicane token with a speed limit, part of the Rallyman DIRT Copilot Pack.
42    ChicaneWithLimit(Terrain),
43    /// Jump token, part of the Rallyman DIRT Copilot Pack.
44    Jump(Terrain),
45    /// Water token, part of the Rallyman DIRT Copilot Pack.
46    Water(Terrain),
47    /// Ascent tile modifier, part of the Climb expansion for Rallyman DIRT.
48    ClimbAscent,
49    /// Descent tile modifier, part of the Climb expansion for Rallyman DIRT.
50    ClimbDescent,
51    /// Cloud tile modifier, part of the Climb expansion for Rallyman DIRT.
52    Cloud,
53    /// Oxygen tile modifier, part of the Climb expansion for Rallyman DIRT.
54    Oxygen(u8),
55    /// Joker entrance arch, part of the RX expansion for Rallyman DIRT.
56    JokerEntrance,
57    /// Joker exit arch, part of the RX expansion for Rallyman DIRT.
58    JokerExit,
59    /// Finishing line arch, part of the RX expansion for Rallyman DIRT.
60    Finish,
61    /// Shortcut mud spray, part of the Rallyman DIRT core box.
62    MudSpray,
63}
64
65impl TokenId {
66    /// Creates a copy of the current token with inner terrain changed to a
67    /// given value.
68    ///
69    /// # Examples
70    ///
71    /// ```
72    /// # use rekee::token::*;
73    /// # use rekee::tile::Terrain;
74    /// let token = TokenId::Chicane(Terrain::Asphalt);
75    /// let other_token = token.with_terrain(Terrain::Snow);
76    /// assert_eq!(other_token, TokenId::Chicane(Terrain::Snow));
77    /// ```
78    pub fn with_terrain(&self, terrain: Terrain) -> Self {
79        match self {
80            TokenId::Chicane(_) =>
81                TokenId::Chicane(terrain),
82            TokenId::ChicaneWithLimit(_) =>
83                TokenId::ChicaneWithLimit(terrain),
84            TokenId::Jump(_) =>
85                TokenId::Jump(terrain),
86            TokenId::Water(_) =>
87                TokenId::Water(terrain),
88            _ =>
89                *self
90        }
91    }
92
93    #[deprecated(note = "Please use TokenId::base() instead")]
94    pub fn without_terrain(&self) -> Self {
95        self.with_terrain(Terrain::None)
96    }
97
98    /// Creates a copy of the current token with inner oxygen number changed to
99    /// the given value.
100    pub(crate) fn with_number(&self, number: u8) -> Self {
101        match self {
102            TokenId::Oxygen(_) =>
103                TokenId::Oxygen(number),
104            _ =>
105                *self
106        }
107    }
108
109    /// Creates a copy of the current token without terrain info or oxygen level.
110    pub fn base(&self) -> Self {
111        match self {
112            TokenId::Chicane(_) =>
113                TokenId::Chicane(Terrain::None),
114            TokenId::ChicaneWithLimit(_) =>
115                TokenId::ChicaneWithLimit(Terrain::None),
116            TokenId::Jump(_) =>
117                TokenId::Jump(Terrain::None),
118            TokenId::Water(_) =>
119                TokenId::Water(Terrain::None),
120            TokenId::Oxygen(_) =>
121                TokenId::Oxygen(0),
122            _ =>
123                *self
124        }
125    }
126
127    /// Iterator over all track tokens.
128    ///
129    /// Tokens with terrain use the neutral `Terrain::None` value.
130    ///
131    /// # Examples
132    ///
133    /// ```
134    /// # use rekee::token::*;
135    /// let tokens: Vec<String> = TokenId::iter()
136    ///     .map(|val| val.to_string())
137    ///     .collect();
138    /// assert_eq!(tokens, vec![
139    ///     "Chicane",
140    ///     "Chicane Limit",
141    ///     "Jump",
142    ///     "Water",
143    ///     "Climb Ascent",
144    ///     "Climb Descent",
145    ///     "Cloud",
146    ///     "Oxygen",
147    ///     "Joker Entrance",
148    ///     "Joker Exit",
149    ///     "Finish",
150    ///     "Mud Spray"]);
151    /// ```
152    pub fn iter() -> std::slice::Iter<'static, Self> {
153        const TOKENS: [TokenId; 12] = [
154            TokenId::Chicane(Terrain::None),
155            TokenId::ChicaneWithLimit(Terrain::None),
156            TokenId::Jump(Terrain::None),
157            TokenId::Water(Terrain::None),
158            TokenId::ClimbAscent,
159            TokenId::ClimbDescent,
160            TokenId::Cloud,
161            TokenId::Oxygen(0),
162            TokenId::JokerEntrance,
163            TokenId::JokerExit,
164            TokenId::Finish,
165            TokenId::MudSpray,
166        ];
167        TOKENS.iter()
168    }
169
170    /// Edition that this token belongs to.
171    ///
172    /// Each token appears only within a single edition, this is different from
173    /// [TileInfo::editions()](crate::tile::TileInfo::editions).
174    ///
175    /// # Examples
176    ///
177    /// ```
178    /// # use rekee::edition::Edition;
179    /// # use rekee::tile::Terrain;
180    /// # use rekee::token::TokenId;
181    /// let token = TokenId::Jump(Terrain::Asphalt);
182    /// assert_eq!(token.edition(), Edition::DirtCopilotPack);
183    ///
184    /// let token = TokenId::JokerEntrance;
185    /// assert_eq!(token.edition(), Edition::DirtRx);
186    /// ```
187    pub fn edition(&self) -> Edition {
188        match self {
189            TokenId::Chicane(_) =>
190                Edition::DirtCopilotPack,
191            TokenId::ChicaneWithLimit(_) =>
192                Edition::DirtCopilotPack,
193            TokenId::Jump(_) =>
194                Edition::DirtCopilotPack,
195            TokenId::Water(_) =>
196                Edition::DirtCopilotPack,
197            TokenId::ClimbAscent =>
198                Edition::DirtClimb,
199            TokenId::ClimbDescent =>
200                Edition::DirtClimb,
201            TokenId::Cloud =>
202                Edition::DirtClimb,
203            TokenId::Oxygen(_) =>
204                Edition::DirtClimb,
205            TokenId::JokerEntrance =>
206                Edition::DirtRx,
207            TokenId::JokerExit =>
208                Edition::DirtRx,
209            TokenId::Finish =>
210                Edition::DirtRx,
211            TokenId::MudSpray =>
212                Edition::DirtCoreBox,
213        }
214    }
215}

223impl fmt::Display for TokenId {
224    fn fmt(&self, fmt: &mut fmt::Formatter) -> fmt::Result {
225        let mut terrain = Terrain::None;
226        match self {
227            TokenId::Chicane(val) => {
228                terrain = *val;
229                write!(fmt, "Chicane")?;
230            },
231            TokenId::ChicaneWithLimit(val) => {
232                terrain = *val;
233                write!(fmt, "Chicane Limit")?;
234            },
235            TokenId::Jump(val) => {
236                terrain = *val;
237                write!(fmt, "Jump")?;
238            },
239            TokenId::Water(val) => {
240                terrain = *val;
241                write!(fmt, "Water")?;
242            },
243            TokenId::ClimbAscent =>
244                write!(fmt, "Climb Ascent")?,
245            TokenId::ClimbDescent =>
246                write!(fmt, "Climb Descent")?,
247            TokenId::Cloud =>
248                write!(fmt, "Cloud")?,
249            TokenId::Oxygen(val) => {
250                write!(fmt, "Oxygen")?;
251                if *val > 0 {
252                    write!(fmt, " {}", val)?;
253                }
254            }
255            TokenId::JokerEntrance =>
256                write!(fmt, "Joker Entrance")?,
257            TokenId::JokerExit =>
258                write!(fmt, "Joker Exit")?,
259            TokenId::Finish =>
260                write!(fmt, "Finish")?,
261            TokenId::MudSpray =>
262                write!(fmt, "Mud Spray")?,
263        };
264        if terrain != Terrain::None {
265            write!(fmt, " {}", terrain)?;
266        }
267        Ok(())
268    }

280impl FromStr for TokenId {
281    type Err = ParseTokenError;
282
283    fn from_str(val: &str) -> Result<Self, Self::Err> {
284        // match strings from both trait implementations, std::fmt::Display and serde::Serialize
285        let mut s = val.replace(char::is_whitespace, "-");
286        s.make_ascii_lowercase();
287
288        let mut name = s.as_ref();
289        let mut terrain = Terrain::None;
290        let mut number = 0_u8;
291        if let Some((prefix, suffix)) = s.rsplit_once('-') {
292            if let Ok(val) = suffix.parse() {
293                terrain = val;
294                name = prefix;
295            }
296            if let Ok(val) = suffix.parse() {
297                number = val;
298                name = prefix;
299            }
300        }
301
302        match name {
303            "chicane" =>
304                Ok(TokenId::Chicane(terrain)),
305            "chicane-limit" =>
306                Ok(TokenId::ChicaneWithLimit(terrain)),
307            "jump" =>
308                Ok(TokenId::Jump(terrain)),
309            "water" =>
310                Ok(TokenId::Water(terrain)),
311            "climb-ascent" =>
312                Ok(TokenId::ClimbAscent),
313            "climb-descent" =>
314                Ok(TokenId::ClimbDescent),
315            "cloud" =>
316                Ok(TokenId::Cloud),
317            "oxygen" =>
318                Ok(TokenId::Oxygen(number)),
319            "joker-entrance" =>
320                Ok(TokenId::JokerEntrance),
321            "joker-exit" =>
322                Ok(TokenId::JokerExit),
323            "finish" =>
324                Ok(TokenId::Finish),
325            "mud-spray" =>
326                Ok(TokenId::MudSpray),
327            _ =>
328                Err(ParseTokenError::UnknownName(name.to_string())),
329        }
330    }
331}

334pub enum ParseTokenError {
335    UnknownName(String),
336}

338impl fmt::Display for ParseTokenError {
339    fn fmt(&self, fmt: &mut fmt::Formatter) -> fmt::Result {
340        match self {
341            ParseTokenError::UnknownName(val) =>
342                write!(fmt, "Unknown token name \"{}\"", val),
343        }
344    }
345}

415/// Utility methods on any list of [`TokenId`]s.
416pub trait TokenList {
417
418    /// Returns a summary about the tokens that are contained within the current
419    /// list of tokens.
420    ///
421    /// For each token the entry contains the number of instances that occur in
422    /// the current list of tiles. Note that terrain information is ignored when
423    /// building the token summary. The returned array is always sorted by token
424    /// identifier.
425    ///
426    /// # Examples
427    ///
428    /// ```
429    /// # use rekee::tile::Terrain;
430    /// # use rekee::token::{TokenId, TokenList};
431    /// let tokens = vec![
432    ///     TokenId::JokerEntrance,
433    ///     TokenId::Chicane(Terrain::Gravel),
434    ///     TokenId::Chicane(Terrain::Asphalt),
435    ///     TokenId::Oxygen(1),
436    /// ];
437    /// for row in tokens.token_summary() {
438    ///     println!("{}x token {}", row.count, row.token);
439    /// }
440    /// ```
441    fn token_summary(&self) -> Vec<TokenSummary>;
442
443    /// Returns a summary about the editions that are necessary to build the
444    /// current list of tokens.
445    ///
446    /// For each edition the entry contains the number of tokens that belong to
447    /// the edition.
448    ///
449    /// # Examples
450    ///
451    /// ```
452    /// # use rekee::tile::Terrain;
453    /// # use rekee::token::{TokenId, TokenList};
454    /// let tokens = vec![
455    ///     TokenId::JokerEntrance,
456    ///     TokenId::Chicane(Terrain::Gravel),
457    ///     TokenId::Chicane(Terrain::Asphalt),
458    ///     TokenId::Oxygen(1),
459    /// ];
460    /// for row in tokens.edition_summary() {
461    ///     println!("{}: {} tokens", row.edition, row.token_count);
462    /// }
463    /// ```
464    fn edition_summary(&self) -> Vec<EditionSummary>;
465}
466
467impl<T: AsRef<TokenId> + Clone> TokenList for [T] {
468    fn token_summary(&self) -> Vec<TokenSummary> {
469        let mut summary = BTreeMap::new();
470        for token in self.iter() {
471            let base_token = token.as_ref()
472                .with_terrain(Terrain::None);
473            let entry = summary.entry(base_token)
474                .or_insert(0);
475            *entry += 1;
476        }
477        let summary: Vec<_> = summary.iter()
478            .map(|(&token, &count)|
479                TokenSummary::new(token, count)
480            )
481            .collect();
482        summary
483    }
484
485    fn edition_summary(&self) -> Vec<EditionSummary> {
486        let mut summary = BTreeMap::new();
487        for token in self.iter() {
488            let edition = token.as_ref().edition();
489            let entry = summary.entry(edition)
490                .or_insert(0);
491            *entry += 1;
492        }
493        let summary: Vec<_> = summary.iter()
494            .map(|(&edition, &count)|
495                EditionSummary::new(edition, count)
496            )
497            .collect();
498        summary
499    }
500}
