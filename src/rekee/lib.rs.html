1//----------------------------------------------------------------------------
2//! Library providing track editor functions for the Rekee web application.
3//!
4//! When the library is compiled into a WASM module it will include a `main`
5//! function which connects the view components to the web page environment.
6//!
7//! The following HTML elements are looked up by element id:
8//! * `catalog-container` (mandatory)
9//! * `catalog-config-container` (mandatory)
10//! * `export-container` (mandatory)
11//! * `map-container` (mandatory)
12//! * `track-info-container` (optional)
13//! * `map-config-container` (optional)
14//! * `map-detail-container` (optional)
15//! * `version` (optional)
16//! * `welcome` (optional)
17//!
18//! The library can also be used outside the web environment as a native
19//! library to process track files. For details take a look at the files in the
20//! `examples` folder.

72#[doc(hidden)]
73#[wasm_bindgen]
74pub fn main() -> Result<(), JsValue> {
75    let window = web_sys::window().unwrap();
76    let document = window.document().unwrap();
77    let layout = Layout::new(Orientation::pointy(), Point(60.0, 60.0), Point(540.0, 400.0));
78
79    let parent = document.get_element_by_id("catalog-container")
80        .ok_or("Cannot find '#catalog-container' parent element for catalog component")?;
81    CatalogController::init(CatalogView::new(parent, &layout)?);
82
83    let parent = document.get_element_by_id("map-container")
84        .ok_or("Cannot find '#map-container' parent element for map component")?;
85    MapController::init(MapView::new(parent, &layout)?);
86
87    let parent = document.get_element_by_id("catalog-config-container")
88        .ok_or("Cannot find '#catalog-config-container' parent element for catalog config component")?;
89    CatalogConfigController::init(CatalogConfigView::new(parent)?);
90
91    if let Some(parent) = document.get_element_by_id("track-info-container") {
92        TrackInfoController::init(TrackInfoView::new(parent)?);
93    }
94
95    if let Some(parent) = document.get_element_by_id("map-config-container") {
96        MapConfigController::init(MapConfigView::new(parent)?);
97    }
98
99    if let Some(parent) = document.get_element_by_id("map-detail-container") {
100        MapDetailController::init(MapDetailView::new(parent, &layout)?);
101    }
102
103    let parent = document.get_element_by_id("export-container")
104        .ok_or("Cannot find '#export-container' parent element for export component")?;
105    ExportController::init(ExportView::new(parent, &layout)?);
106
107    if let Some(parent) = document.get_element_by_id("version") {
108        VersionView::new(parent)?;
109    }
110
111    // build some example track when compiling in development mode
112    if cfg!(debug_assertions) {
113        nuts::publish(InsertTileEvent { id: tile!(102, b), pos: (0, 0).into(), dir: Direction::D });
114        nuts::publish(AppendTileEvent { id: tile!(104, b), pos: Some((-1, 0).into()), hint: None });
115        nuts::publish(AppendTileEvent { id: tile!(113, b), pos: None, hint: "R".parse().ok() });
116        nuts::publish(AppendTileEvent { id: tile!(117, b), pos: None, hint: "r".parse().ok() });
117        nuts::publish(AppendTileEvent { id: tile!(114, b), pos: None, hint: "R".parse().ok() });
118        nuts::publish(AppendTileEvent { id: tile!(115, b), pos: None, hint: None });
119        nuts::publish(AppendTileEvent { id: tile!(115, b), pos: None, hint: None });
120        nuts::publish(AppendTileEvent { id: tile!(108, b), pos: None, hint: "r".parse().ok() });
121        nuts::publish(AppendTileEvent { id: tile!(110, b), pos: None, hint: "R".parse().ok() });
122        nuts::publish(AppendTileEvent { id: tile!(107, b), pos: None, hint: None });
123        nuts::publish(InsertTileEvent { id: tile!(101), pos: (0, -2).into(), dir: Direction::D });
124        nuts::publish(UpdateSelectedTileEvent { pos: Coordinate::new(-1, 0).to_pixel(&layout) });
125        let token = PlacedToken::new(TokenId::ChicaneWithLimit(Terrain::Asphalt), (0.0, 0.0).into(), 3.0.into());
126        nuts::publish(AddSelectedTileTokenEvent { token });
127        let token = PlacedToken::new(TokenId::Chicane(Terrain::Asphalt), (0.32, 0.0).into(), 0.0.into());
128        nuts::publish(AddSelectedTileTokenEvent { token });
129        let token = PlacedToken::new(TokenId::Chicane(Terrain::Asphalt), (-0.32, 0.0).into(), 0.0.into());
130        nuts::publish(AddSelectedTileTokenEvent { token });
131        nuts::publish(UpdateSelectedTileEvent { pos: Coordinate::new(1, -1).to_pixel(&layout) });
132        let token = PlacedToken::new(TokenId::Finish, (0.5, 0.0).into(), 3.0.into());
133        nuts::publish(AddSelectedTileTokenEvent { token });
134        nuts::publish(AlignCenterEvent);
135        nuts::publish(UpdateTitleEvent { title: String::from("Short Track") });
136        nuts::publish(UpdateAuthorEvent { author: String::from("CarmLima") });
137    }
138
139    // initialize after all other controllers are completed to ignore events during setup
140    if let Some(parent) = document.get_element_by_id("welcome") {
141        WelcomeController::init(WelcomeView::new(parent)?);
142    }
143
144    // restore previous view settings
145    nuts::publish(LoadSettingsEvent {});
146
147    Ok(())
148}
