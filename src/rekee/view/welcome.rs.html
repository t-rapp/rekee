22#[derive(Default, Serialize, Deserialize)]
23#[serde(default)]
24pub struct WelcomeSettings {
25    pub hidden: bool,
26}

29
30pub struct WelcomeView {
31    inner: Element,
32    header: Element,
33    close_cb: Closure<dyn Fn(web_sys::Event)>,
34    catalog_gt_button: Option<Element>,
35    catalog_dirt_button: Option<Element>,
36    catalog_all_button: Option<Element>,
37    select_catalog_cb: Closure<dyn Fn(web_sys::Event)>,
38}
39
40impl WelcomeView {
41    pub fn new(parent: Element) -> Result<Self> {
42        let document = parent.owner_document().unwrap();
43        let inner = parent;
44
45        let header = inner.query_selector(".message-header")?
46            .ok_or("Cannot find header of welcome message element")?;
47        inner.set_hidden(false);
48
49        let close_cb = Closure::wrap(Box::new(move |_event: web_sys::Event| {
50            nuts::publish(HideWelcomeEvent);
51        }) as Box<dyn Fn(_)>);
52
53        header.add_event_listener_with_callback("click",
54            close_cb.as_ref().unchecked_ref())?;
55        header.class_list().add_1("is-clickable")?;
56
57        let select_catalog_cb = Closure::wrap(Box::new(move |event: web_sys::Event| {
58            let button = check!(event.current_target()
59                .and_then(|target| target.dyn_into::<web_sys::HtmlElement>().ok()));
60            let series: Option<Series> = button.get_attribute("data-value")
61                .and_then(|val| val.parse().ok());
62            let edition_iter = if let Some(series) = series {
63                series.editions()
64            } else {
65                Edition::iter()
66            };
67            let editions: Vec<Edition> = edition_iter.copied()
68                    .collect();
69            nuts::publish(UpdateCatalogEditionsEvent { editions });
70        }) as Box<dyn Fn(_)>);
71
72        let catalog_gt_button = document.get_element_by_id("select-catalog-gt-button");
73        if let Some(ref button) = catalog_gt_button {
74            button.add_event_listener_with_callback("click", select_catalog_cb.as_ref().unchecked_ref())?;
75        }
76
77        let catalog_dirt_button = document.get_element_by_id("select-catalog-dirt-button");
78        if let Some(ref button) = catalog_dirt_button {
79            button.add_event_listener_with_callback("click", select_catalog_cb.as_ref().unchecked_ref())?;
80        }
81
82        let catalog_all_button = document.get_element_by_id("select-catalog-all-button");
83        if let Some(ref button) = catalog_all_button {
84            button.add_event_listener_with_callback("click", select_catalog_cb.as_ref().unchecked_ref())?;
85        }
86
87        Ok(WelcomeView {
88            inner, header, close_cb, catalog_gt_button, catalog_dirt_button,
89            catalog_all_button, select_catalog_cb
90        })
91    }
92
93    pub fn load_settings(&mut self, settings: &WelcomeSettings) {
94        self.inner.set_hidden(settings.hidden);
95    }
96
97    pub fn save_settings(&mut self) -> WelcomeSettings {
98        let hidden = self.inner.hidden();
99        WelcomeSettings { hidden }
100    }
101
102    pub fn set_hidden(&self, value: bool) {
103        debug!("update welcome hidden: {}", value);
104        self.inner.set_hidden(value);
105
106        // remember visibility state
107        nuts::send_to::<WelcomeController, _>(SaveSettingsEvent);
108    }
109}
110
111impl Drop for WelcomeView {
112    fn drop(&mut self) {
113        let _ = self.header.remove_event_listener_with_callback("click",
114            self.close_cb.as_ref().unchecked_ref());
115        if let Some(ref button) = self.catalog_gt_button {
116            let _ = button.remove_event_listener_with_callback("click",
117                self.select_catalog_cb.as_ref().unchecked_ref());
118        }
119        if let Some(ref button) = self.catalog_dirt_button {
120            let _ = button.remove_event_listener_with_callback("click",
121                self.select_catalog_cb.as_ref().unchecked_ref());
122        }
123        if let Some(ref button) = self.catalog_all_button {
124            let _ = button.remove_event_listener_with_callback("click",
125                self.select_catalog_cb.as_ref().unchecked_ref());
126        }
127    }
128}
