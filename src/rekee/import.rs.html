23type Result<T> = std::result::Result<T, ImportError>;
24
25#[derive(Debug)]
26pub enum ImportError {
27    UnknownTileId(String),
28    UnknownTokenId(String),
29    UnsupportedTokenId(String),
30    JsonError(serde_json::Error),
31}
32
33impl fmt::Display for ImportError {
34    fn fmt(&self, fmt: &mut fmt::Formatter) -> fmt::Result {
35        match self {
36            ImportError::UnknownTileId(val) =>
37                write!(fmt, "Unknown tile identifier \"{}\"", val),
38            ImportError::UnknownTokenId(val) =>
39                write!(fmt, "Unknown token identifier \"{}\"", val),
40            ImportError::UnsupportedTokenId(val) =>
41                write!(fmt, "Unsupported token identifier \"{}\"", val),
42            ImportError::JsonError(err) =>
43                write!(fmt, "Error processing JSON data: {}", err),
44        }
45    }
46}
47
48impl From<serde_json::Error> for ImportError {
49    fn from(error: serde_json::Error) -> Self {
50        ImportError::JsonError(error)
51    }
52}

69    #[derive(Debug, Default, Deserialize, Serialize)]
70    struct ImportRoot {
71        #[serde(default)]
72        title: String,
73        #[serde(default, skip_serializing_if = "String::is_empty")]
74        author: String,
75        tiles: Vec<ImportTile>,
76    }
77
78    #[derive(Debug, Default, Deserialize, Serialize)]
79    struct ImportTile {
80        q: i32,
81        r: i32,
82        id: String,
83        dir: i32,
84        #[serde(default, skip_serializing_if = "Vec::is_empty")]
85        tokens: Vec<ImportToken>,
86    }
87
88    #[derive(Debug, Default, Deserialize, Serialize)]
89    struct ImportToken {
90        id: String,
91        q: f32,
92        r: f32,
93        dir: f32,
94    }
95
96    impl TryInto<PlacedToken> for &ImportToken {
97        type Error = ImportError;
98
99        fn try_into(self) -> Result<PlacedToken> {
100            let id = self.id.parse::<TokenId>()
101                .map_err(|_| ImportError::UnknownTokenId(self.id.clone()))?;
102            let pos = FloatCoordinate::new(self.q, self.r);
103            let dir = FloatDirection(self.dir);
104            Ok(PlacedToken::new(id, pos, dir))
105        }
106    }
107
108    impl From<&PlacedToken> for ImportToken {
109        fn from(value: &PlacedToken) -> ImportToken {
110            let id = format!("{:x}", value.id);
111            let q = value.pos.q();
112            let r = value.pos.r();
113            let dir = f32::from(value.dir);
114            ImportToken { id, q, r, dir }
115        }
116    }
117
118    pub fn import(data: &str) -> Result<Map> {
119        let data: ImportRoot = serde_json::from_str(data)?;
120
121        let mut map = Map::new();
122        if !data.title.is_empty() {
123            map.set_title(&data.title);
124        }
125        if !data.author.is_empty() {
126            map.set_author(&data.author);
127        }
128        for tile in data.tiles {
129            let pos = Coordinate::new(tile.q, tile.r);
130            let id = tile.id.parse::<TileId>()
131                .map_err(|_| ImportError::UnknownTileId(tile.id.clone()));
132            let dir = Direction::from(tile.dir);
133            let tokens = tile.tokens.iter()
134                // tokens that have no corresponding import identifier will be silently dropped here
135                .filter_map(|token| token.try_into().ok())
136                .collect();
137            if let Ok(id) = id {
138                map.insert_with_tokens(id, pos, dir, tokens);
139            }
140        }
141
142        Ok(map)
143    }
144
145    pub fn export(map: &Map) -> Result<String> {
146        let mut data = ImportRoot {
147            title: map.title().to_string(),
148            author: map.author().to_string(),
149            tiles: Vec::with_capacity(map.tiles().len()),
150        };
151
152        for tile in map.tiles() {
153            let q = tile.pos.q();
154            let r = tile.pos.r();
155            let id = format!("{:x}", tile.id());
156            let dir = i32::from(tile.dir);
157            let tokens = tile.tokens.iter()
158                .map(|token| token.into())
159                .collect();
160            data.tiles.push(ImportTile { q, r, id, dir, tokens });
161        }
162
163        serde_json::to_string(&data)
164            .map_err(ImportError::JsonError)
165    }
166}

358    const IMPORT_OFFSET: Coordinate = Coordinate::new(-81, 52);
359
360    pub fn import(data: &str) -> Result<Map> {
361        let data: ImportRoot = serde_json::from_str(data)?;
362
363        let mut map = Map::new();
364        if !data.title.is_empty() {
365            map.set_title(&data.title);
366        }
367        if !data.author.is_empty() {
368            map.set_author(&data.author);
369        }
370        for tile in data.tiles {
371            // convert from offset to axial coordinates
372            let q = -tile.x - ((tile.y & 1) + tile.y) / 2;
373            let r = tile.y;
374            let pos = Coordinate::new(q, r) - IMPORT_OFFSET;
375            let id = tile.id.parse::<TileId>()
376                // special mapping for pit-stop tile
377                .map(|id| if id.num() == 101 { id.base() } else { id })
378                .unwrap_or_default();
379            let dir = Direction::from(tile.dir);
380            let tokens = tile.tokens.iter()
381                .filter_map(|token| token.try_into().ok())
382                .collect();
383            debug!("import tile ({}, {}), {}, {} -> {}, {}, {}",
384                tile.x, tile.y, tile.id, tile.dir, pos, id, dir);
385            map.insert_with_tokens(id, pos, dir, tokens);
386        }
387
388        Ok(map)
389    }

391    pub fn export(map: &Map) -> Result<String> {
392        // generate some default values for data that is not used in our editor
393        let mut data = ImportRoot {
394            path: "C:\\Applications\\RallymanGT Track Editor\\My Tracks".to_string(),
395            title: map.title().to_string(),
396            author: map.author().to_string(),
397            tiles: Vec::with_capacity(map.tiles().len()),
398            low_res_width: 2480,
399            low_res_height: 1748,
400        };
401
402        let token_layout = token_layout();
403        for tile in map.tiles() {
404            let pos = tile.pos + IMPORT_OFFSET;
405            // convert from axial to offset coordinates
406            let x = -pos.q() - ((pos.r() & 1) + pos.r()) / 2;
407            let y = pos.r();
408            let mut id = match tile.id().num() {
409                // special mapping for pit-stop tile
410                101 => "101a".to_string(),
411                _ => tile.id().to_string(),
412            };
413            if let Some(info) = TileInfo::get(tile.id()) {
414                if info.series() == Series::Dirt {
415                    // DIRT tiles are handled uppercase
416                    id.make_ascii_uppercase();
417                }
418            }
419            let dir = i32::from(tile.dir);
420            debug!("export tile {}, {}, {} -> ({}, {}), {}, {}",
421                tile.pos, tile.id(), tile.dir, x, y, &id, dir);
422
423            let mut tokens = Vec::with_capacity(tile.tokens.len());
424            for token in &tile.tokens {
425                let pos = token.pos.to_pixel(&token_layout);
426                let x = pos.x();
427                let y = pos.y();
428                let dir = token.dir.to_angle();
429                let id = token.id.try_into()?;
430                tokens.push(ImportToken { x, y, dir, id });
431            }
432
433            data.tiles.push(ImportTile { x, y, dir, id, tokens });
434        }
435
436        serde_json::to_string(&data)
437            .map_err(ImportError::JsonError)
438    }
439
440    fn token_layout() -> Layout {
441        // invert the x-axis, keep the y-axis
442        Layout::new(Orientation::pointy(), Point(-1.0, 1.0), Point(0.0, 0.0))
443    }

448/// Import map data with format auto-detection.
449pub fn import_auto(data: &str) -> Result<Map> {
450    // try our native format first
451    let result = native::import(data);
452    if result.is_ok() {
453        return result;
454    }
455    // then try other formats
456    if let Ok(map) = rgt::import(data) {
457        return Ok(map);
458    }
459    // no format succeeded, return the first error
460    result
461}

463/// Helper function to generate a valid filename from a (map title) string.
464pub fn build_file_name(name: &str) -> String {
465    let mut result: String = name.trim().chars()
466        // remove all control characters
467        .filter(|chr| !chr.is_control())
468        // replace path separators
469        .map(|chr| if chr != '/' && chr != '\\' { chr } else { '_' })
470        // remove special characters on Microsoft Windows
471        .filter(|chr| *chr != '<' && *chr != '>' && *chr != ':' &&
472                      *chr != '"' && *chr != '|' && *chr != '?' && *chr != '*')
473        .collect();
474    if result.is_empty() {
475        result.push_str("MyTrack");
476    }
477    result
478}
